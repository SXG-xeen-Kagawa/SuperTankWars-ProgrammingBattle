#region MOVE

        [Header("<Move (移動)>")]
        [SerializeField, Tooltip("基本となる移動の強さ"), Range(0.1f,1.0f)] float m_movePower = 1.0f;


        void DoMove()
        {
            // 近い戦車を探す 
            int nearestId = FindTankIdWithMinDistance();
            if (0 < nearestId)
            {
                // 一番近い戦車 
                var nearestTank = m_allTanksInfo[nearestId];

                // そちらへ移動 
                Vector3 toGoal = nearestTank.Position - m_nowPosition;
                SetCaterpillarTowardWorldDir(m_nowRotation, toGoal.normalized, forwardPower: m_movePower);

                // 目標座標をデバッグ表示
                SXG_DebugDrawPositionMarker(nearestTank.Position);

            } else
            {
                // 近い戦車が見つからないなら何もしない 
            }
        }

#endregion
